!SLIDE commandline
# New project #

    $ mkdir drnic-hello-world
    $ cd drnic-hello-world
    $ touch README.md
    $ git init
    $ git add .
    $ git commit -m "Initial commit"

!SLIDE commandline
# Push to GitHub #

    $ gh create-from-local
    $ git config remote.origin.url
    git@github.com:drnic/drnic-hello-world.git

    $ git push origin master
